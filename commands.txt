# Initialize CSV file
"Date,UserName,File,Message" | Out-File -Encoding utf8 commits.csv

# Temporary buffer for filenames
$filenameBuffer = @()
$commitBuffer = @()

# Fetch log details
git log --pretty=format:'"%ci","%an","%s"' --name-only | ForEach-Object {
    # If the line is empty, we are at the end of a commit's details
    if ($_ -eq "") {
        # Join filenames with semicolons
        $files = ($filenameBuffer -join ";")
        # Add to commit buffer
        $commitBuffer += "$commitDetails,""$files"""
        # Clear the filename buffer
        $filenameBuffer = @()
    }
    # If the line matches the commit format, capture the commit details
    elseif ($_ -match "^[0-9]{4}-[0-9]{2}-[0-9]{2} ") {
        $commitDetails = $_
    }
    # Otherwise, add the line to the filename buffer
    else {
        $filenameBuffer += $_
    }
}

# Sort by date and write to CSV
$commitBuffer | Sort-Object | ForEach-Object {
    $_ | Out-File -Append -Encoding utf8 commits.csv
}
